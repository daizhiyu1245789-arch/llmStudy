## 01_把 tinyllm 跑起来

### 训练命令

```bash
python -m tinyllm.train \
  --data_path data/sample.txt \
  --max_steps 50 \
  --log_interval 10 \
  --eval_interval 20 \
  --save_interval 20
```

### 生成命令

```bash
python -m tinyllm.generate \
  --ckpt_path checkpoints/latest.pt \
  --prompt "今天" \
  --max_new_tokens 80 \
  --temperature 0.8
```

### temperature 对比

```text
temperature=0.3：更稳定，重复概率更高。
temperature=1.2：更随机，变化更多，也更容易乱。
```

## 04_Node 集成 Hugging Face

### 普通生成 curl

```bash
curl -X POST http://localhost:3001/api/generate \
  -H "Content-Type: application/json" \
  -d '{"prompt":"Deep learning is","maxNewTokens":80}'
```

### 流式生成 curl

```bash
curl -N -X POST http://localhost:3001/api/generate/stream \
  -H "Content-Type: application/json" \
  -d '{"prompt":"What is deep learning?","maxNewTokens":80}'
```

### 现象参考

```text
普通生成：等完整结果回来。
流式生成：一段一段返回。
HF_TOKEN 错误：模型调用失败。
prompt 为空：Zod 参数校验失败。
```

## 05_Prompt 模板和结构化输出

### buildExtractPrompt

```ts
export function buildExtractPrompt(text: string): string {
  return [
    "你是一个文本信息抽取助手。",
    "",
    "任务：",
    "从用户输入中提取结构化信息。",
    "",
    "规则：",
    "- 只输出 JSON",
    "- 不要输出 Markdown",
    "- 不要添加解释",
    "- 如果字段不存在，用空数组或空字符串",
    "",
    "输出 JSON 格式：",
    '{ "title": "string", "summary": "string", "keywords": ["string"], "todoItems": ["string"] }',
    "",
    "用户输入：",
    text,
  ].join("\n");
}
```

### Zod 校验

```ts
import { z } from "zod";

const ExtractResultSchema = z.object({
  title: z.string(),
  summary: z.string(),
  keywords: z.array(z.string()),
  todoItems: z.array(z.string()),
});

export function stripMarkdownFence(text: string): string {
  return text
    .trim()
    .replace(/^```json\s*/i, "")
    .replace(/^```\s*/i, "")
    .replace(/\s*```$/i, "")
    .trim();
}

export function parseExtractResult(output: string) {
  const clean = stripMarkdownFence(output);
  const json = JSON.parse(clean);
  return ExtractResultSchema.parse(json);
}
```

## 06_Embedding 和向量检索

### cosine similarity

```ts
function cosineSimilarity(a: number[], b: number[]): number {
  let dot = 0;
  let normA = 0;
  let normB = 0;

  for (let i = 0; i < a.length; i++) {
    dot += a[i] * b[i];
    normA += a[i] * a[i];
    normB += b[i] * b[i];
  }

  return dot / (Math.sqrt(normA) * Math.sqrt(normB));
}
```

### 内存检索

```ts
const docs = [
  { content: "Node SSE 流式输出", embedding: [1, 0] },
  { content: "晚饭吃火锅", embedding: [0, 1] },
];

const query = [0.9, 0.1];

const results = docs
  .map((doc) => ({
    ...doc,
    score: cosineSimilarity(query, doc.embedding),
  }))
  .sort((a, b) => b.score - a.score);

console.log(results[0].content); // Node SSE 流式输出
```

## 07_RAG 文档问答

### buildRagPrompt

```ts
type RagContext = {
  title: string;
  source: string;
  content: string;
  score: number;
};

export function buildRagPrompt(question: string, contexts: RagContext[]) {
  const contextText = contexts
    .map((item, index) =>
      [
        `[${index + 1}] 标题：${item.title}`,
        `来源：${item.source}`,
        `相关度：${item.score.toFixed(3)}`,
        "内容：",
        item.content,
      ].join("\n")
    )
    .join("\n\n");

  return [
    "你是一个严谨的文档问答助手。",
    "请只根据参考资料回答。",
    "如果资料中没有答案，请说“资料中没有找到相关信息”。",
    "回答关键结论时标注来源编号，比如 [1]。",
    "",
    "参考资料：",
    contextText || "无",
    "",
    "用户问题：",
    question,
  ].join("\n");
}
```

## 08_Tool Calling 工具调用

### getWeather schema + handler

```ts
import { z } from "zod";

const GetWeatherArgsSchema = z.object({
  city: z.string().min(1).max(50),
});

async function getWeather(args: { city: string }) {
  return {
    city: args.city,
    weather: "晴",
    temperature: "26°C",
  };
}

export async function runGetWeather(rawArgs: unknown) {
  const args = GetWeatherArgsSchema.parse(rawArgs);
  return getWeather(args);
}
```

### tool registry

```ts
type ToolHandler = (args: unknown) => Promise<unknown>;

const toolRegistry: Record<string, ToolHandler> = {
  getWeather: runGetWeather,
};

export async function runTool(name: string, args: unknown) {
  const handler = toolRegistry[name];

  if (!handler) {
    throw new Error(`Unknown tool: ${name}`);
  }

  return handler(args);
}
```

### 模拟 action

```ts
const action = {
  type: "tool_call",
  toolName: "getWeather",
  arguments: {
    city: "杭州",
  },
};

const result = await runTool(action.toolName, action.arguments);
console.log(result);
```

## 09_日志评估和部署

### requestId + latency

```ts
import crypto from "node:crypto";

app.post("/api/generate", async (req, res, next) => {
  const requestId = crypto.randomUUID();
  const startedAt = Date.now();

  try {
    const result = await generateText(req.body);

    console.log({
      requestId,
      model: "gpt2",
      latencyMs: Date.now() - startedAt,
      status: "success",
    });

    res.json({ requestId, result });
  } catch (error) {
    console.error({
      requestId,
      latencyMs: Date.now() - startedAt,
      status: "error",
      error,
    });

    next(error);
  }
});
```
